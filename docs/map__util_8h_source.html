111  vec_Veci<Dim> rayTrace(const Vecf<Dim> &pt1, const Vecf<Dim> &pt2) {
112  Vecf<Dim> diff = pt2 - pt1;
113  decimal_t k = 0.8;
114  int max_diff = (diff / res_).template lpNorm<Eigen::Infinity>() / k;
115  decimal_t s = 1.0 / max_diff;
116  Vecf<Dim> step = diff * s;
117 
118  vec_Veci<Dim> pns;
119  Veci<Dim> prev_pn = Veci<Dim>::Constant(-1);
120  for (int n = 1; n < max_diff; n++) {
121  Vecf<Dim> pt = pt1 + step * n;
122  Veci<Dim> new_pn = floatToInt(pt);
123  if (isOutside(new_pn))
124  break;
125  if (new_pn != prev_pn)
126  pns.push_back(new_pn);
127  prev_pn = new_pn;
128  }
129  return pns;
130  }
